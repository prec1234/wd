<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Collect values
    $name = $_POST['name'];
    $email = $_POST['email'];
    $age = $_POST['age'];

    // Display values
    echo "<h2>Form Data Received</h2>";
    echo "Name: " . htmlspecialchars($name) . "<br>";
    echo "Email: " . htmlspecialchars($email) . "<br>";
    echo "Age: " . htmlspecialchars($age) . "<br>";
}
?>
